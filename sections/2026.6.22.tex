\subsection*{はじめまして}

\begin{studygoal}[今日のメモ]
  记录初次见面时使用的自我介绍句型。
\end{studygoal}

\begin{vocab}{あいさつ}
  \vocabentry{私}{わたし}{代词}{我}
  \vocabentry{学生}{がくせい}{名词}{学生}
  \vocabentry{日本語}{にほんご}{名词}{日语}
  \vocabentry{よろしく}{よろしく}{副词}{请多关照}
\end{vocab}

\begin{grammar}{A は B です}
  用来表达“A 是 B”。助词 \term{は} 提示主题，句末的
  \term{です} 使语气礼貌。
\end{grammar}

\begin{example}
  \sentence{私は学生です。}
    {わたし は がくせい です。}
    {我是学生。}
\end{example}

\begin{pitfall}[读音提醒]
  主题助词 \jp{「は」} 读作 \jp{「わ」}。生词可以写成
  \furigana{漢字}{かんじ} 的形式标注假名。
\end{pitfall}
